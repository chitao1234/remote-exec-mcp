# Windows XP Daemon Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a standalone Windows XP daemon that speaks the existing broker-facing `/v1/*` RPC contract over plain HTTP, plus the broker-side insecure transport opt-in needed to route to it.

**Architecture:** Keep the current Rust broker and modern Rust daemon intact, then add a separate C++ XP daemon project under `crates/` that is built with `i686-w64-mingw32-g++`. The broker learns one new per-target transport choice, while the XP daemon implements only the reduced v1 feature set: `cmd.exe`-backed exec sessions without PTY, in-process `apply_patch`, single-file-only transfer, and no image support.

**Tech Stack:** Rust 2024, Tokio, reqwest, axum test fixtures, C++17, Win32 API, `i686-w64-mingw32-g++`, header-only `cpp-httplib`, header-only `nlohmann/json`, `make`

// crates/remote-exec-broker/src/config.rs

#[derive(Debug, Clone, Deserialize)]
pub struct TargetConfig {
    pub base_url: String,
    #[serde(default)]
    pub ca_pem: Option<PathBuf>,
    #[serde(default)]
    pub client_cert_pem: Option<PathBuf>,
    #[serde(default)]
    pub client_key_pem: Option<PathBuf>,
    #[serde(default)]
    pub allow_insecure_http: bool,
    pub expected_daemon_name: Option<String>,
}

impl TargetConfig {
    fn scheme(&self) -> anyhow::Result<&'static str> {
        if self.base_url.starts_with("https://") {
            return Ok("https");
        }
        if self.base_url.starts_with("http://") {
            return Ok("http");
        }
        anyhow::bail!("target base_url must start with http:// or https://");
    }

    fn validate_transport(&self, name: &str) -> anyhow::Result<()> {
        match self.scheme()? {
            "http" => {
                anyhow::ensure!(
                    self.allow_insecure_http,
                    "target `{name}` uses http://; http:// targets require allow_insecure_http = true"
                );
            }
            "https" => {
                anyhow::ensure!(self.ca_pem.is_some(), "target `{name}` is missing ca_pem");
                anyhow::ensure!(
                    self.client_cert_pem.is_some(),
                    "target `{name}` is missing client_cert_pem"
                );
                anyhow::ensure!(
                    self.client_key_pem.is_some(),
                    "target `{name}` is missing client_key_pem"
                );
            }
            _ => unreachable!(),
        }
        Ok(())
    }
}

impl BrokerConfig {
    pub(crate) fn validate(&self) -> anyhow::Result<()> {
        anyhow::ensure!(
            !self.targets.contains_key("local"),
            "configured target name `local` is reserved for broker-host filesystem access"
        );
        for (name, target) in &self.targets {
            target.validate_transport(name)?;
        }
        Ok(())
    }
}

// crates/remote-exec-broker/src/daemon_client.rs

impl DaemonClient {
    pub async fn new(config: &TargetConfig) -> anyhow::Result<Self> {
        let client = if config.base_url.starts_with("http://") {
            anyhow::ensure!(
                config.allow_insecure_http,
                "http:// targets require allow_insecure_http = true"
            );
            reqwest::Client::builder()
                .build()
                .context("building insecure daemon client")?
        } else {
            let ca_pem = config
                .ca_pem
                .as_ref()
                .context("ca_pem is required for https targets")?;
            let client_cert_pem = config
                .client_cert_pem
                .as_ref()
                .context("client_cert_pem is required for https targets")?;
            let client_key_pem = config
                .client_key_pem
                .as_ref()
                .context("client_key_pem is required for https targets")?;

            let ca = Certificate::from_pem(&tokio::fs::read(ca_pem).await?)?;
            let identity = Identity::from_pem(
                &[
                    tokio::fs::read(client_cert_pem).await?,
                    tokio::fs::read(client_key_pem).await?,
                ]
                .concat(),
            )?;

            reqwest::Client::builder()
                .use_rustls_tls()
                .add_root_certificate(ca)
                .identity(identity)
                .build()
                .context("building daemon client")?
        };

        Ok(Self {
            client,
            base_url: config.base_url.clone(),
        })
    }
}

// crates/remote-exec-daemon-xp/src/config.cpp

#include <fstream>
#include <map>
#include <stdexcept>
#include "config.h"

static std::string trim(const std::string& raw) {
    const std::string whitespace = " \t\r\n";
    const std::size_t start = raw.find_first_not_of(whitespace);
    if (start == std::string::npos) {
        return "";
    }
    const std::size_t end = raw.find_last_not_of(whitespace);
    return raw.substr(start, end - start + 1);
}

DaemonConfig load_config(const std::string& path) {
    std::ifstream input(path.c_str());
    if (!input) {
        throw std::runtime_error("unable to open config file: " + path);
    }

    std::map<std::string, std::string> values;
    std::string line;
    while (std::getline(input, line)) {
        line = trim(line);
        if (line.empty() || line[0] == ';' || line[0] == '#') {
            continue;
        }
        const std::size_t equals = line.find('=');
        if (equals == std::string::npos) {
            throw std::runtime_error("invalid config line: " + line);
        }
        values[trim(line.substr(0, equals))] = trim(line.substr(equals + 1));
    }

    DaemonConfig config;
    config.target = values.at("target");
    config.listen_host = values.at("listen_host");
    config.listen_port = std::atoi(values.at("listen_port").c_str());
    config.default_workdir = values.at("default_workdir");
    return config;
}

// crates/remote-exec-daemon-xp/src/server.cpp

#include <windows.h>
#include "common.h"
#include "http_helpers.h"
#include "server.h"

int run_server(const DaemonConfig& config) {
    httplib::Server server;
    const std::string daemon_instance_id = std::to_string(GetTickCount());

    server.Post("/v1/health", [&](const httplib::Request&, httplib::Response& res) {
        write_json(res, Json{
            {"status", "ok"},
            {"daemon_version", REMOTE_EXEC_XP_VERSION},
            {"daemon_instance_id", daemon_instance_id}
        });
    });

    server.Post("/v1/target-info", [&](const httplib::Request&, httplib::Response& res) {
        write_json(res, Json{
            {"target", config.target},
            {"daemon_version", REMOTE_EXEC_XP_VERSION},
            {"daemon_instance_id", daemon_instance_id},
            {"hostname", "windows-xp"},
            {"platform", "windows"},
            {"arch", "x86"},
            {"supports_pty", false},
            {"supports_image_read", false}
        });
    });

    server.Post("/v1/image/read", [&](const httplib::Request&, httplib::Response& res) {
        write_rpc_error(res, 400, "image_unsupported", "image read is not supported on this target");
    });

    return server.listen(config.listen_host.c_str(), config.listen_port) ? 0 : 1;
}

static std::string read_available(HANDLE pipe) {
    DWORD available = 0;
    if (!PeekNamedPipe(pipe, NULL, 0, NULL, &available, NULL) || available == 0) {
        return "";
    }

    std::string buffer;
    buffer.resize(available);
    DWORD read = 0;
    if (!ReadFile(pipe, &buffer[0], available, &read, NULL)) {
        return "";
    }
    buffer.resize(read);
    return buffer;
}

Json SessionStore::start_command(
    const std::string& command,
    const std::string& workdir,
    bool tty,
    const std::string& shell
) {
    if (tty) {
        return Json{
            {"code", "tty_unsupported"},
            {"message", "tty is not supported on this host"}
        };
    }

    sa.nLength = sizeof(sa);
    sa.lpSecurityDescriptor = NULL;
    sa.bInheritHandle = TRUE;

    HANDLE stdout_write = NULL;
    HANDLE stdin_read = NULL;
    HANDLE stdin_write = NULL;
    CreatePipe(&stdout_read, &stdout_write, &sa, 0);
    CreatePipe(&stdin_read, &stdin_write, &sa, 0);

    SetHandleInformation(stdin_write, HANDLE_FLAG_INHERIT, 0);

    ZeroMemory(&si, sizeof(si));
    si.cb = sizeof(si);
    si.dwFlags = STARTF_USESTDHANDLES;

    si.hStdOutput = stdout_write;
    si.hStdError = stdout_write;

    PROCESS_INFORMATION pi;
    ZeroMemory(&pi, sizeof(pi));

    std::string resolved_shell = shell.empty() ? "cmd.exe" : shell;
    std::string cmdline = resolved_shell + " /C " + command;
    std::vector<char> mutable_cmdline(cmdline.begin(), cmdline.end());

        NULL,
        &mutable_cmdline[0],
        NULL,
        NULL,
        TRUE,
        0,
        NULL,
        workdir.empty() ? NULL : workdir.c_str(),
        &si,
        &pi
    );
    CloseHandle(stdin_read);
    CloseHandle(stdout_write);

    if (!ok) {
        throw std::runtime_error("CreateProcessA failed");
    }

    std::shared_ptr<LiveSession> session(new LiveSession());
    session->id = make_session_id();
    session->process_info = pi;
    session->stdin_write = stdin_write;
    session->stdout_read = stdout_read;

    session->running = true;

Json SessionStore::write_stdin(const std::string& daemon_session_id, const std::string& chars) {
    std::map<std::string, std::shared_ptr<LiveSession> >::iterator it = sessions_.find(daemon_session_id);
    if (it == sessions_.end()) {
        throw std::runtime_error("unknown_session");
    }

    DWORD written = 0;
    if (!chars.empty()) {
        WriteFile(it->second->stdin_write, chars.data(), static_cast<DWORD>(chars.size()), &written, NULL);
    }

    const std::string output = read_available(it->second->stdout_read);
    DWORD exit_code = STILL_ACTIVE;
    GetExitCodeProcess(it->second->process_info.hProcess, &exit_code);

    return Json{
        {"daemon_session_id", daemon_session_id},
        {"running", exit_code == STILL_ACTIVE},
        {"chunk_id", "xp002"},
        {"wall_time_seconds", 0.0},
        {"exit_code", exit_code == STILL_ACTIVE ? Json(nullptr) : Json(static_cast<int>(exit_code))},
        {"original_token_count", output.size()},
        {"output", output},
        {"warnings", Json::array()}
    };
}

#include <filesystem>
#include <fstream>
#include <sstream>
#include <stdexcept>
#include <vector>
#include "patch_engine.h"

namespace fs = std::filesystem;

struct PatchAction {
    enum Kind { Add, Delete, Update } kind;
    std::string path;
    std::string move_to;
    std::vector<std::string> body;
};

static std::string slurp(const fs::path& path) {
    std::ifstream input(path.c_str(), std::ios::binary);
    if (!input) {
        throw std::runtime_error("unable to read " + path.string());
    }
    return std::string((std::istreambuf_iterator<char>(input)), std::istreambuf_iterator<char>());
}

static void write_atomically(const fs::path& path, const std::string& content) {
    const fs::path temp = path.string() + ".tmp";
    std::ofstream output(temp.c_str(), std::ios::binary);
    output << content;
    output.close();
    fs::rename(temp, path);
}

static std::vector<PatchAction> parse_patch(const std::string& patch_text) {
    std::vector<PatchAction> actions;
    std::istringstream input(patch_text);
    std::string line;
    PatchAction* current = NULL;

    while (std::getline(input, line)) {
        if (line.rfind("*** Add File: ", 0) == 0) {
            actions.push_back(PatchAction{PatchAction::Add, line.substr(14), "", {}});
            current = &actions.back();
        } else if (line.rfind("*** Delete File: ", 0) == 0) {
            actions.push_back(PatchAction{PatchAction::Delete, line.substr(17), "", {}});
            current = &actions.back();
        } else if (line.rfind("*** Update File: ", 0) == 0) {
            actions.push_back(PatchAction{PatchAction::Update, line.substr(17), "", {}});
            current = &actions.back();
        } else if (line.rfind("*** Move to: ", 0) == 0 && current != NULL) {
            current->move_to = line.substr(13);
        } else if ((line.size() > 0 && (line[0] == '+' || line[0] == '-' || line[0] == ' ')) && current != NULL) {
            current->body.push_back(line);
        }
    }

    return actions;
}

PatchApplyResult apply_patch(const std::string& root, const std::string& patch_text) {
    const std::vector<PatchAction> actions = parse_patch(patch_text);
    std::vector<std::string> summary;

    for (std::size_t i = 0; i < actions.size(); ++i) {
        const PatchAction& action = actions[i];
        fs::path path = fs::path(root) / action.path;

        if (action.kind == PatchAction::Add) {
            std::string content;
            for (std::size_t line = 0; line < action.body.size(); ++line) {
                if (!action.body[line].empty() && action.body[line][0] == '+') {
                    content += action.body[line].substr(1) + "\n";
                }
            }
            fs::create_directories(path.parent_path());
            write_atomically(path, content);
            summary.push_back("A " + action.path);
        } else if (action.kind == PatchAction::Delete) {
            fs::remove(path);
            summary.push_back("D " + action.path);
        } else {
            const std::string old_text = slurp(path);
            std::string new_text = old_text;
            for (std::size_t line = 0; line < action.body.size(); ++line) {
                if (!action.body[line].empty() && action.body[line][0] == '-') {
                    const std::string needle = action.body[line].substr(1) + "\n";
                    const std::size_t offset = new_text.find(needle);
                    if (offset == std::string::npos) {
                        throw std::runtime_error("patch context not found in " + action.path);
                    }
                    new_text.erase(offset, needle.size());
                } else if (!action.body[line].empty() && action.body[line][0] == '+') {
                    new_text += action.body[line].substr(1) + "\n";
                }
            }

            const fs::path destination = action.move_to.empty() ? path : fs::path(root) / action.move_to;
            fs::create_directories(destination.parent_path());
            write_atomically(destination, new_text);
            if (!action.move_to.empty()) {
                fs::remove(path);
            }
            summary.push_back("M " + (action.move_to.empty() ? action.path : action.move_to));
        }
    }

    std::string output = "Success. Updated the following files:\n";
    for (std::size_t i = 0; i < summary.size(); ++i) {
        output += summary[i] + "\n";
    }
    return PatchApplyResult{output};
}

namespace fs = std::filesystem;

static std::string slurp_binary(const fs::path& path) {
    std::ifstream input(path.c_str(), std::ios::binary);
    if (!input) {
        throw std::runtime_error("transfer source missing");
    }
    return std::string((std::istreambuf_iterator<char>(input)), std::istreambuf_iterator<char>());
}

ExportedFile export_file(const std::string& absolute_path) {
    const fs::path path(absolute_path);
    if (!path.is_absolute()) {
        throw std::runtime_error("transfer path is not absolute");
    }
    if (!fs::exists(path) || !fs::is_regular_file(path)) {
        throw std::runtime_error("transfer source must be a regular file");
    }
    return ExportedFile{"file", slurp_binary(path)};
}

ImportSummary import_file(const std::string& bytes, const std::string& absolute_path, bool replace_existing) {
    const fs::path path(absolute_path);
    if (!path.is_absolute()) {
        throw std::runtime_error("transfer path is not absolute");
    }
    const bool existed = fs::exists(path);
    if (existed && !replace_existing) {
        throw std::runtime_error("destination path already exists");
    }
    fs::create_directories(path.parent_path());
    std::ofstream output(path.c_str(), std::ios::binary | std::ios::trunc);
    output.write(bytes.data(), static_cast<std::streamsize>(bytes.size()));
    return ImportSummary{"file", static_cast<std::uint64_t>(bytes.size()), 1, 0, existed};
}
